This code compiles:

struct Parser<'c, 's> {
    context: &'c &'s str,
}

    fn parse(&self) -> &'s str {
        self.context
    }
}

fn parse_context<'s>(context: &'s str) -> &'s str {
    Parser { context: &context }.parse()
}
