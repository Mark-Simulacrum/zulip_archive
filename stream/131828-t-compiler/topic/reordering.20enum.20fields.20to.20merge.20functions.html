Is rustc allowed to reorder enum fields?

I found that for code like this

pub enum A {
    True,
    False
}

pub fn af(a: bool) -> char {
    if a {
        'X'
    } else {
        'F'
    }
}

pub fn bf(a: A) -> char {
    match a {
        A::True => 'X',
        A::False => 'F',
    }
}
