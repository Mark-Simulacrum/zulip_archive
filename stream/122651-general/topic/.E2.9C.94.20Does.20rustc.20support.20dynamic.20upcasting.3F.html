So, the following snippet works on nightly:

#![feature(trait_upcasting)]

use ::core::any::{self, Any};

trait MyAnyWithTypeName : Any {
    fn type_name (self: &'_ Self)
      -> &'static str
    {
        impl<T : Any> MyAnyWithTypeName for T
        {}

        any::type_name::<Self>()
    }
}

fn check (p: &dyn MyAnyWithTypeName)
{
    eprintln!("About to downcast something of type `{}`", p.type_name());
    dbg!(<dyn Any>::is::<i32>(p));
}

fn main ()
{
    check(&());
    check(&42);
}
