use criterion::{black_box, criterion_group, criterion_main, Criterion};
use core::cmp::Ordering;

#[inline(never)]
pub fn cmp_u8(a: &[u8], b: &[u8]) -> Ordering {
    a.cmp(b)
}

#[inline(never)]
pub fn cmp_usize(a: &[usize], b: &[usize]) -> Ordering {
    a.cmp(b)
}

fn criterion_benchmark(c: &mut Criterion) {
    let a = [453342564535342usize; 4096];
    let aa: [u8; 4096*8] = unsafe {std::mem::transmute(a)};

    c.bench_function("cmp_u8", |b| {
        b.iter(|| black_box(cmp_u8(&aa, &aa)))
    });
    c.bench_function("cmp_usize", |b| {
        b.iter(|| black_box(cmp_usize(&a, &a)))
    });
}

criterion_group!(benches, criterion_benchmark);

Note that the align_to-twice version given is incorrect -- it'll give different prefix and suffix lengths if the arrays are differently aligned, and thus will compare incorrectly for those cases.
