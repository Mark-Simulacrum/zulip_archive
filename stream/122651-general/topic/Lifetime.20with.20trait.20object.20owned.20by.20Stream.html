// This should help reduce complexity. Instead of keeping track of a paginator's,
// state with multiple fields, there are three states. The value of the `Error` state
// will be taken and returned, and the state changed to `Closed`.
enum RequestStreamState<'f, R, T, E> {
    Pending(Pin<Box<dyn Future<Output = Result<R, E>> + 'f>>),
    Ready(T),
    Error(E),
    Closed,
}

pub struct SearchResultsPaginator<'p> {
    // This paginator needs to have the same lifetime of the `Client`.
    client: &'p Client,
    // The paginator needs to own the parameters because it mutates on the first request,
    // and we want to avoid the callee needing to wrap this with a smart pointer.
    params: SearchParams,
    // The initial offset of the `SearchParams` passed to the constructor.
    offset: usize,
    // This keeps track of internal state.
    state: RequestStreamState<'p, SearchResults, VecDeque<ProjectResult>, surf::Error>,
    // This is not the total number of projects from the API,
    // rather it is how many this iterator will yield.
    count: usize,
    // This is an option because the value is only present if the first request succeeded.
    total: Option<usize>,
}

impl<'p> SearchResultsPaginator<'p> {
    pub async fn new(client: &'p Client, mut params: SearchParams) -> SearchResultsPaginator<'p> {
        let (status, total);
        let offset = params.offset.unwrap_or(0);

        match Self::first_request(client, &mut params).await {
            Ok(search) => {
                status = RequestStreamState::Ready(search.hits);
                // The API returns the total number of projects that met the search criteria.
                // ...
                total = Some(search.total_hits - offset);
            }
            Err(error) => {
                status = RequestStreamState::Error(error);
                total = None;
            }
        };

        Self {
            client,
            params,
            offset,
            state: status,
            count: 0,
            total,
        }
    }

    async fn first_request(
        client: &Client,
        params: &mut SearchParams,
    ) -> surf::Result<SearchResults> {
        // This is a hack to get the API to provide the correct total result count.
        // The API seems to only return the correct number of projects when the initial limit is one.
        let limit = std::mem::replace(&mut params.limit, Some(1));
        let result = client.search(params).await;
        params.limit = limit;

        result
    }
}

impl Stream for SearchResultsPaginator<'_> {
    type Item = surf::Result<ProjectResult>;

    fn poll_next(self: Pin<&mut Self>, context: &mut Context<'_>) -> Poll<Option<Self::Item>> {
        // Unpin `self` so that mutation is possible.
        let this: &mut _ = self.get_mut();

        match &mut this.state {
            // At some point in the past, a new request was made. Check to see if it has delivered
            // a result yet, if it has, and handle accordingly. Otherwise, we simply return `Pending`.
            RequestStreamState::Pending(future) => match future.poll(context) {
                // The future returned from the last page requested has resolved.
                // Determine what to do with it, there are several situations.
                Poll::Ready(result) => match result {
                    // It is successful, so take the `ProjectResult`s and
                    // save them to the current state.
                    Ok(mut search) => {
                        this.state = RequestStreamState::Ready(search.hits);
                        // Since the state has been changed, try to return a `ProjectResult`.
                        this.poll_next(context)
                    }
                    // When an error is encountered after the first result,
                    // we can pass the error to the callee directly,
                    // rather than going through an iteration where the status is `Error`.
                    // Also set `self.state` to `Closed`.
                    Err(error) => {
                        this.state = RequestStreamState::Closed;
                        Poll::Ready(Some(Err(error)))
                    }
                },
                // Still waiting, just forward that state.
                Poll::Pending => Poll::Pending,
            },
            // There is already a queue. If it is empty, request the next page.
            RequestStreamState::Ready(projects) => match projects.pop_front() {
                // Increment the counter and return this `ProjectResult`.
                Some(project) => {
                    this.count += 1;

                    // It is expected that if this is reached, the first page
                    // of projects has been retrieved successfully. Unwrapping should be safe here.
                    if this.count >= this.total.unwrap() {
                        // There are no more projects expected.
                        this.state = RequestStreamState::Closed;
                    }

                    Poll::Ready(Some(Ok(project)))
                }
                // Request the next page.
                None => {
                    // Here an `if let` must be used because the state cannot be `PartialEq`.
                    if let RequestStreamState::Closed = this.state {
                        // If an error has occurred this should already be
                        // handled by the `Closed` branch. The same is true if there
                        // are no more results, as the state is in the `Some` branch.
                        unreachable!()
                    } else {
                        // It must be time to request another page, because we have not closed.
                        // Here the search offset is set to the sum of `self.offset` (which is the initial offset)
                        // and `self.count` to get an absolute position at the start of the next page.
                        this.params.offset = Some(this.offset + this.count);
                        this.state =
                            RequestStreamState::Pending(Box::pin(this.client.search(&this.params)));

                        Poll::Pending
                    }
                }
            },
            // This should generally only occur if the first request set the state to `Error`.
            // If this happens in the middle of iteration, the error will have already been
            // forwarded to the callee and the `self.state` set to `Closed`.
            RequestStreamState::Error(_) => {
                // Necessary to replace the value because it can't be moved.
                // Set `self.status` to `Closed` and return `Ready` with the error.
                let RequestStreamState::Error(error) =
                    std::mem::replace(&mut this.state, RequestStreamState::Closed);

                // Forward the error, the state has been set to `Closed`.
                Poll::Ready(Some(Err(error)))
            }
            // Either we have consumed all the results from all the pages, or the previous call
            // returned an `Error`.
            RequestStreamState::Closed => Poll::Ready(None),
        }
    }

    fn size_hint(&self) -> (usize, Option<usize>) {
        (0, self.total)
    }
}
