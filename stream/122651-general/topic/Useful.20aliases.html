Aman Arora (Dec 04 2020 at 04:07):

This is the sort of the minimized version of what I use. Push and pop just allow me to not have to go to rust root to run commands :sweat_smile:

runRustStuff()
{
    pushd $RUST_PROJECT_REPO
    nice ./x.py fmt -j32
    nice ./x.py $@
    popd
}

buildRust()
{
    runRustStuff build -j50 -i --stage 1 $@
}

testRust()
{
    runRustStuff test -i --stage 1 $@
}

rustCheck()
{
    pushd $SG_RUST_PROJECT_REPO
    ./x.py check -j32
    popd
}
