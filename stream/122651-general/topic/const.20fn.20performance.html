I wrote this small sample that takes between 7-10 seconds on my machine for a 1 MB array copy in a const fn, and I think the problem I had is basically the same thing.

#![feature(const_eval_limit)]
#![const_eval_limit = "10000000"]
fn main() {
    const MB: usize = 1024 * 1024;
    const L: usize = MB;
    const DATA: [u8; L] = [3; L];
    const TEST: [u8; L] = test(&DATA);

    const fn test(data: &[u8]) -> [u8; L] {
        let mut ret = [0u8; L];
        let mut i = 0;
        loop {
            if i >= data.len() {
                break;
            }
            ret[i] = data[i];
            i = i + 1;
        }
        ret
    }
    println!("{}", TEST[1024]);
}
