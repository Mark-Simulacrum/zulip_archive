Like:

#[cfg(miri)]
use pdep_bmi2 as pdep;

#[cfg(not(miri))]
use core::arch::x86_64::_pdep_u64 as pdep;

pub const fn pdep_bmi2(v: u64, mask: u64) -> u64 {
    let mut mask = mask;
    let mut src = v;
    let mut dst = 0;
    let mut i = u64::BITS;
    // for i in (0..u64::BITS).rev() {
    loop {
        i -= 1;
        let b = mask & 1;
        dst |= (src & b);
        dst = dst.rotate_right(1);
        src >>= b;

        if i == 0 {
            break;
        }
        mask >>= 1;
    }
    dst
}
