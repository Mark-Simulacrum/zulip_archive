/// Parse [Ipv4Addr] from [core::str]. Each octet must be 1-3 base-10 ascii digits separated by '.' char.
///
/// Enabling Strict-mode forbids leading zeros and trailing dot.
///
/// IP addresses in non-base-10 formats such as hex or octal are rejected.
///
/// Inputs that are parsed successfully in Strict-mode MUST produce identical result when STRICT is false.
///
/// License: CC0, Public Domain, etc
const fn parse_ip4<const STRICT: bool>(ip: &str) -> Option<Ipv4Addr> {
    if STRICT {
        const IP_MAX_LEN: usize = 15; // "255.255.255.255".len()
        const IP_MIN_LEN: usize = 7; // "0.0.0.0".len()
        if ip.len() < IP_MIN_LEN || ip.len() > IP_MAX_LEN {
            return None;
        }
    } else {
        const IP_MAX_LEN: usize = 16; // "255.255.255.255.".len()
        const IP_MIN_LEN: usize = 7; // "0.0.0.0".len()
        if ip.len() < IP_MIN_LEN || ip.len() > IP_MAX_LEN {
            return None;
        }
    }
    let mut r = [0u8; 4];
    let mut current_octet = 0;
    let mut char_count = 0;
    let mut fail = false;
    let mut is_previous_dot = true;
    let mut expect_dot = false;
    let mut reader = 0;

    let mut working_octet = 0u8; // Using this temp variable is a lot faster than r[current_octet].

    let ip = ip.as_bytes();

    while reader < ip.len() {
        let b = ip[reader];
        reader += 1;
        if b.is_ascii_digit() && (!STRICT || !expect_dot) {
            let u = b - b'0'; // ASCII to u8

            if STRICT && u == 0 && working_octet == 0 {
                // In strict-mode, each leading zero MUST be followed by either '.' char or EOF
                expect_dot = true;
            }

            let (v, flag) = working_octet.overflowing_mul(10);
            fail |= flag;
            let (v, flag) = v.overflowing_add(u);
            fail |= flag;
            working_octet = v;
            is_previous_dot = false;
            char_count += 1;
            if char_count == 4 {
                fail = true;
            }
        } else if b == b'.' {
            // The address MUST NOT contain two consecutive '.' chars
            fail |= is_previous_dot;
            is_previous_dot = true;
            r[current_octet] = working_octet;
            working_octet = 0;
            current_octet += 1;
            char_count = 0;
            if STRICT {
                expect_dot = false; // Reset flag
            }
            // In Strict-mode, the address MUST NOT contain a trailing '.' char
            if current_octet == 4 && (STRICT || reader != ip.len()) {
                fail = true;
                current_octet = 0;
            }
        } else {
            // Byte is neither ASCII number nor '.'
            fail = true;
        }
    }
    if current_octet == 3 {
        r[3] = working_octet;
    }
    if fail {
        None
    } else {
        Some(Ipv4Addr::new(r[0], r[1], r[2], r[3]))
    }
}
