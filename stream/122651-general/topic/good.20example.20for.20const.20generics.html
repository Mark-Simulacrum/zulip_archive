fn collect_to_array<T, const N: usize>(mut it: impl Iterator<Item = T>) -> [T; N]
where
    T: Copy + Default,
{
    let mut array = [T::default(); N];
    for (i, x) in it.by_ref().enumerate() {
        array[i] = x;
    }
    assert!(it.next().is_none());
    array
}

fn main() {
    let [a, b, c, d, e, f] = collect_to_array(0..6);
    dbg!(a, b, c, d, e, f);
}

HeroicKatora (Jul 12 2020 at 15:33):

An other thing entirely would be to show const-generics for indexing a slice as an array:
The most amazing thing about this is that generic argument deduction works!

pub struct RangeTo<const N: usize>;

impl<T, const N: usize> Index<RangeTo<N>> for [T] {
    type Output = [T; N];
    fn index(&self, _: RangeTo<{ N }>) -> &[T; N] {
        let slice = &self[..N];
        unsafe { &*(slice.as_ptr() as *const [T; N])  }
    }
}

fn with_beginning(slice: &[u8]) {
    let [first, second] = slice[RangeTo];
}
