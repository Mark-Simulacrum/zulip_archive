Chenyang Shao (Dec 04 2021 at 18:18):

Is it practical to set default values when reading environment variables?

pub fn or<K: AsRef<OsStr>, T: FromStr>(k: K, default: T) -> T {
    let v = var(k);

    v.ok().and_then(|s| s.parse::<T>().ok()).unwrap_or(default)
}

usecase:

            PostgresConfig {
                host: or("POSTGRES_HOST", "localhost"),
                port: or("POSTGRES_PORT", 5432),
                user: or("POSTGRES_USER", "postgres"),
                pwd: or("POSTGRES_PASSWORD", "postgres"),
                name: or("POSTGRES_DB", default.name),
            },
