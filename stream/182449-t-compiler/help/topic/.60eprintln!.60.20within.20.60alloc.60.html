extern "C" {
    fn printf(s: *const u8, ...);
}

macro_rules! log {
    ($s:literal, $($arg:tt)*) => {
        unsafe { printf(concat!($s, "\n\0", $($arg)*); }
    }
}

@bjorn3 Thanks! I need to print to stderr, and I couldn't get your code to work, but I ended up with something similar:

extern "C" {
    fn dprintf(fd: i32, s: *const u8, ...);
}

macro_rules! my_printf {
    ($s:literal) => {
        unsafe { dprintf(2, $s as *const u8); }
    }
}

fn main() {
    my_printf!(b"hello world\n\0");
}
