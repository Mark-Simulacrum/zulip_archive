struct Array<T, const N: usize>([T; N]);

impl<T, const N: usize> Array<T, N> {
    pub fn get<const I: usize>(&mut self) -> &mut T {
            &mut self.0[I]
    }
}

fn main() {
    let mut array = [0; 3];
    array[3] = 4;

    let mut array = Array::<u8, 3>([0; 3]);
    *array.get::<3>() = 4;
}
