Emil Gardström (Sep 06 2021 at 19:46):

got it.

let def_id = self.tcx.hir().local_def_id(hir_id);
let def_kind = self.tcx.def_kind(def_id);
let parent_node_hir_id = self.tcx.hir().get_parent_node(hir_id);
let parent = self.tcx.hir().find(parent_node_hir_id);
info!("child: {:?}", def_kind);
info!("parent: {:?}", parent);
match (parent, def_kind) {
    (
        Some(hir::Node::Variant(hir::Variant {
            data: hir::VariantData::Tuple(..), ..
        })),
        def::DefKind::Field,
    ) => { todo!() }
}
