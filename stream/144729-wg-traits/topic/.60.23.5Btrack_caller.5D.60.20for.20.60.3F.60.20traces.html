match Try::into_result(<expr>) {
    Ok(val) => #[allow(unreachable_code)] val,
    Err(err) => #[allow(unreachable_code)] {
        let err = From::from(err);
        // the compiler should insantiate a Location to represent the
        // position of the `?` in the source file
        err.track(<&'static core::panic::Location<'static>>);
        if in_enclosing_try {
            break 'catch_target Try::from_error(err);
        } else {
            return Try::from_error(err);
        }
    }
}
