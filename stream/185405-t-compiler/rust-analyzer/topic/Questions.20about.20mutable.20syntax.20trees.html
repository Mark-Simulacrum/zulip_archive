Other thing is a refactoring that inserts a space for comments after their sigil. Eg. //foo -> // foo, //!bar -> //! bar. I haven't found a way to mutate the tokens directly, but I left this code 6 months ago. Maybe RA API changed since then and it could be simplified?

fn fix(ctx: &mut LintCtx<'_>) {
    let file = ctx.make_mut(ctx.file_ast.clone());
    let descendants = file.syntax().descendants();

    for node in descendants {
        let replacements: Vec<_> = bad_comments(&node)
            .map(|comment| {
                let mut comment_text = comment.text().to_owned();
                comment_text.insert(sigil_end(&comment), ' ');

                // Based on impl in `ra_syntax::ast::make::tokens` module
                let new_comment = ast::SourceFile::parse(&comment_text)
                    .ok()
                    .unwrap()
                    .syntax()
                    .clone_for_update()
                    .first_child_or_token()
                    .unwrap()
                    .into_token()
                    .unwrap();

                (comment, new_comment)
            })
            .collect();

        for (old_comment, new_comment) in replacements {
            let idx = old_comment.syntax().index();
            node.splice_children(idx..idx + 1, vec![new_comment.into()]);
        }
    }
}
