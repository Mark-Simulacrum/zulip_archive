It looks like we can't write a test like this in hir_ty:

trait Tr {
    fn method(&self) -> u8 {}
}

fn outer() {
    ().method();
  //^^^^^^^^^^^ u8

    fn inner() {
        ().method();
      //^^^^^^^^^^^ u8
    }
}
