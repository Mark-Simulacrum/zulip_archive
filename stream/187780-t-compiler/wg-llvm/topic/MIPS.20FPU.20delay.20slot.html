#![no_std]
#![no_main]

#![feature(bench_black_box)]

psp::module!("crash-test",1,0);

    let out = check_branch(core::hint::black_box(0.0));
    psp::dprintln!("{}", out);
}

#[no_mangle]
#[inline(never)]
fn check_branch(dx: f32) -> f32 {
    if dx == 0.0 {
        1.0
    } else {
        2.0
    }
}
