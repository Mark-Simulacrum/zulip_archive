pub fn set_pac(cx: &CodegenCx<'ll, '_>, llfn: &'ll Value) {
    if cx.sess().must_use_pac() {
        llvm::AddFunctionAttrStringValue(
            llfn,
            llvm::AttributePlace::Function,
            const_cstr!("ptrauth-calls"),
            const_cstr!(""),
        );
        llvm::AddFunctionAttrStringValue(
            llfn,
            llvm::AttributePlace::Function,
            const_cstr!("ptrauth-returns"),
            const_cstr!(""),
        );
    }
}
