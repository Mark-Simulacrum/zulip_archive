  • I believe get_array() would be more suitable name. Chunks makes sense for a collection of arrays, get_chunk() returns a single array.
  • Include _unchecked variants. Compiler is not (yet) perfect at figuring safe bounds
  • Make it const by using #![feature(const_slice_index)]
// Actual code:
pub fn get(src: &[u8]) -> Option<(u64, u64)> {
    /* if ... } else*/ if src.len() >= 17 {
        // Read 2 u64 from the end:
        // Temp value used as a workaround for compiler stupidity.
        let temp = src.get(src.len()-16..).expect("Unreachable panic");

        let a = u64::from_le_bytes(temp.get(0..8).expect("Unreachable panic").try_into().expect("Unreachable panic"));
        let b = u64::from_le_bytes(temp.get(8..16).expect("Unreachable panic").try_into().expect("Unreachable panic"));
        Some((a, b))
    } else {
        None
    }
}
