Jacob Pratt said:

ok, in that case, Step can be split into ForwardStep and BidirectionalStep:

pub trait ForwardStep: Clone + PartialOrd {
    fn steps_between(start: &Self, end: &Self) -> Option<usize>;
    fn forward_checked(start: Self, count: usize) -> Option<Self>;
    fn forward(start: Self, count: usize) -> Self { ... }
    unsafe fn forward_unchecked(start: Self, count: usize) -> Self { ... }
}

pub trait BidirectionalStep: ForwardStep {
    fn backward_checked(start: Self, count: usize) -> Option<Self>;
    fn backward(start: Self, count: usize) -> Self { ... }
    unsafe fn backward_unchecked(start: Self, count: usize) -> Self { ... }
}

Jacob Lifshay (Jul 26 2021 at 14:47):

reverse iterator/step can be added later:

#[unstable(feature = "reverse_iter")]
pub trait ReverseStep: Clone + PartialOrd {
    /// get `(start..end).len()`
    /// if `steps_between(&a, &b) == Some(N)`, then
    /// it must be the case that `a == backward(b, N)`
    fn steps_between(start: &Self, end: &Self) -> Option<usize>;
    fn backward_checked(start: Self, count: usize) -> Option<Self>;
    fn backward(start: Self, count: usize) -> Self { ... }
    unsafe fn backward_unchecked(start: Self, count: usize) -> Self { ... }
}

impl<T: BidirectionalStep> ReverseStep for T { ... }

#[unstable(feature = "reverse_iter")]
pub trait ReverseIterator {
    type Item;
    fn next_back(&mut) -> Option<Self::Item>;
    ...
}

impl<T: BidirectionalIterator + ?Sized> ReverseIterator for T { ... }
