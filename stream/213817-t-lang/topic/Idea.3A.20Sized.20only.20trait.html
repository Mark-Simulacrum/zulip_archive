Frank Steffahn (Feb 11 2022 at 13:28):

@Charles Lew I don’t fully understand the code in that other thread, in particular I don’t know why you wouldn’t store a &'a mut dyn AsyncIterator<TS, Item = I> directly in the Next struct, how these TS parameters are supposed to interact with user-implementations, or why there’s a Pin<&mut self> method on a : Unpin trait. Anyways, here’s the best I could come up with modifying your code:

#![feature(ptr_metadata)]
#![feature(adt_const_params)]
#![allow(incomplete_features)]

use std::future::Future;
use std::marker::PhantomData;
use std::pin::Pin;
use std::ptr::{self, DynMetadata};
use std::rc::Rc;
use std::task::Context;
use std::task::Poll;

#[derive(PartialEq, Eq)]
pub enum ThreadSafety {
    NA,
    Send,
    Sync,
    SendAndSync,
}

pub struct Next<'a, I, const TS: ThreadSafety> {
    async_iter_data: *mut (),
    async_iter_metadata: DynMetadata<dyn AsyncIterator<TS, Item = I> + 'a>,
    phantom_data: PhantomData<&'a mut Rc<I>>,
}

impl<'a, I, const TS: ThreadSafety> Future for Next<'a, I, { TS }> {
    type Output = Option<I>;
    fn poll(self: Pin<&mut Self>, ctx: &mut Context<'_>) -> Poll<Option<I>> {
        let iter: *mut dyn AsyncIterator<{ TS }, Item = I> =
            ptr::from_raw_parts_mut(self.async_iter_data, self.async_iter_metadata);
        let iter: &'a mut dyn AsyncIterator<{ TS }, Item = I> = unsafe { &mut *iter };
        Pin::new(iter).poll_next(ctx)
    }
}

unsafe impl<'a, T> Send for Next<'a, T, { ThreadSafety::Send }> {}
unsafe impl<'a, T> Sync for Next<'a, T, { ThreadSafety::Sync }> {}
unsafe impl<'a, T> Send for Next<'a, T, { ThreadSafety::SendAndSync }> {}
unsafe impl<'a, T> Sync for Next<'a, T, { ThreadSafety::SendAndSync }> {}

// unfortunately hard to pull off without splitting the `Item` type into a supertrait
pub trait HasAsyncItertatorItem {
    type Item;
}

// sealed, if you don't want any user-implementations
// of the as_dyn_async_iterator_mut conversion
mod sealed {
    use super::*;
    pub trait Sealed<const TS: ThreadSafety> {}
    impl<const TS: ThreadSafety, T: AsyncIterator<TS>> Sealed<TS> for T {}
}
pub trait AsDynAsyncIteratorMut<const TS: ThreadSafety>: HasAsyncItertatorItem + sealed::Sealed<TS> {
    fn as_dyn_async_iterator_mut(
        &mut self,
    ) -> &mut dyn AsyncIterator<TS, Item = Self::Item>;
}
impl<const TS: ThreadSafety, T: AsyncIterator<TS>> AsDynAsyncIteratorMut<TS> for T {
    fn as_dyn_async_iterator_mut(
        &mut self,
    ) -> &mut dyn AsyncIterator<TS, Item = Self::Item> {
        self
    }
}

pub trait AsyncIterator<const TS: ThreadSafety>: Unpin + AsDynAsyncIteratorMut<TS> {
    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>>;

    fn next(&mut self) -> Next<'_, Self::Item, { TS }> {
        let s: *mut (dyn AsyncIterator<{ TS }, Item = Self::Item> + '_) = self.as_dyn_async_iterator_mut();
        let (async_iter_data, async_iter_metadata) = s.to_raw_parts();
        Next {
            async_iter_data,
            async_iter_metadata,
            phantom_data: PhantomData,
        }
    }
}
